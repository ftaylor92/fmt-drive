sub-ant multiple sub-directory ANT example

Sample Simple ANT File

<?xml version="1.0" encoding="UTF-8"?>









<project name="ManningANT" >
    <property name="common.folder" value="D:/proj/fmt/common/source" />
 
    <path id="common.classpath">
        <fileset dir="${common.folder}/lib">
               <include name="*.jar"/>
        </fileset>
    </path>
 
    <target name="init" description="creates folders for build">
        <mkdir dir="build/classes" />
        <mkdir dir="dist" />
    </target>









    <target name="compile" depends="init" description="compiles Java">
        <ant dir="./com" />    <!-- compiles subproject-->
        <subant buildpath="./" /> 
        <subant dir="./**" />
        <javac srcdir="src/" destdir="build/classes/">
            <classpath>
                <pathelement path="sqlj/lib/translator.jar" />
            </classpath>
        </javac>
    </target>









    <target name="doc" depends="init" description="generates JavaDoc">
        <javadoc destdir="build/classes" sourcefiles="fava/*.java">
        </javadoc>
    </target>









    <target name="deploy" depends="compile,doc" description="deploys compiled library">
        <jar destfile="dist/project.jar" basedir="build/classes" />
        <!--FTP -->
    </target>









    <target name="run" depends="compile" description="runs the Java application">
        <java classname="fava.firstCIFA" classpath="build/classes">
            <arg value="a" />
        </java>
    </target>









    <target name="sqlj" description="runs SQL-J pre-compiler over sqlj files">
        <java classname="sqlj.tools.Sqlj" args="src/*.sqlj" fork="yes" failonerror="yes">
            <classpath>
                <pathelement location="C:/oracle/ora92/sqlj/lib/translator.jar"/>
                <pathelement location="C:/oracle/ora92/sqlj/lib/runtime12.jar"/>
                <pathelement location="C:/oracle/ora92/jdbc/lib/classes12.jar"/>
            </classpath>
        </java>
    </target>









    <target name="zip" description="Zips up all the code">
        <zip destfile="website.zip">
               <zipfileset dir="H:/usr/hero" includes="build.xml" prefix="website" />
               <zipfileset dir="H:/usr/hero/dd" prefix="website/deploydescriptors" />
        </zip>
    </target>









    <target name="cvs" description="SubVersion=cvs-like tasks">
        <svn javahl="false">
           <createRepository path="H:/usr/CVSROOT"/>
        </svn>
    </target>
 
    <target name="run-style" depends="buildJar">
        <java classname="com.puppycrawl.tools.checkstyle.gui.Main" 
               classpathref="style.classpath" fork="true">
               <arg value="-c sun_checks.xml -r ." />
               <classpath>
                       <pathelement location="${build}/style.jar"/>
               </classpath >
        </java>
    </target>
 
        <target name="format-java" unless="disabled"
               description="TRY-CATCH: check-out all java source files, format them">
 
               <trycatch>
               <try>
                       <antcall target="eq-java"/>
                       <antcall target="co-java"/>
                       <antcall target="format-java"/>
                       <antcall target="java"/>
               </try>
               <catch>
                       <antcall target="java.undo"/>
               </catch>
               </trycatch>
        </target>
 
        <target name="clean-all" depends="props,clean" description="calls clean target on all subprojects">
               <delete dir="${common-build}"/>
                <antcall target="local" inheritAll="true">
                   <param name="subproject-target" value="clean"/>
                </antcall>
        </target>
 
 
<!-- -------------------JBOSS------------------ Targets-->
  <target name="buildWAR" depends="init,build"
   description="Create the WAR file">
        <mkdir dir="${jar}" />
 
        <war destfile="${jar}/${warFile}" webxml="${dd}/web.xml">
           <fileset dir="${web}" />
           <fileset dir="${web}/DHTML" />
           <fileset dir="${web}/form" />
           <fileset dir="${web}/JSP">
           </fileset>
           <fileset dir="${dd}">
               <include name="*.properties" />
           </fileset>
 
           <lib dir="${lib}" />
 
           <classes dir="${buildDir}">
               <include name="**/*.class" />
               <exclude name="com/ejbean/**" />
           </classes>
 
           <metainf dir="${dd}" includes="jboss-web.xml" />
           <metainf dir="${dd}" includes="jboss.xml" />
           <metainf dir="${dd}" includes="context.xml" />
           <metainf dir="${dd}" includes="jbosscmp-jdbc.xml" />
           <metainf dir="${dd}" includes="jaws.xml" />
           <metainf dir="${dd}" includes="jboss-client.xml" />
           <metainf dir="${dd}" includes="faces-config.xml" />
           
        </war>
  </target>
 
  <target name="buildJAR" depends="init,build"
   description="Create the JAR EJB file">
     <jar destfile="${jar}/${jarFile}" basedir="${buildDir}" includes="com/Calculator.class,com/ejbean/**">
        <metainf dir="${dd}" includes="ejb-jar.xml,jboss.xml,faces-config.xml" />
     </jar>
     <fileset dir="${dd}">
        <include name="*.properties" />
     </fileset>
  </target>
 
  <target name="buildEAR" depends="init,buildWAR"
   description="Create the EAR file">
 
    <delete file="${jar}/${earFile}" />
 
    <ear destfile="${jar}/${earFile}" appxml="${dd-rel}/application.xml">
        <fileset dir="${jar}" includes="*.jar,*.war" />
    </ear>
  </target>
</project>

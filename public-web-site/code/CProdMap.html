void	trimString(char *s)
{
char *firstNonBlank = s;
char *currChar,*lastNonBlank;
	if(firstNonBlank == NULL) return;
	lastNonBlank = firstNonBlank + strlen(firstNonBlank);
	while(*firstNonBlank == ' ' || *firstNonBlank == '\t'){firstNonBlank++;}
	if(firstNonBlank == lastNonBlank){
		*s = 0;
		return;
	}
	currChar = firstNonBlank;
//	while(*--lastNonBlank == ' ' && (lastNonBlank > firstNonBlank));
	lastNonBlank--;
	while((*lastNonBlank == ' ' || *lastNonBlank == '\t') && (lastNonBlank > firstNonBlank)){lastNonBlank--;}
	short len = lastNonBlank - firstNonBlank + 1;
	memcpy(s,firstNonBlank,len);
	s[len] = 0;
}

Boolean IterFindProducer(CProducerRef *aRef, RefInfo *info)
{
	Boolean	exists(FALSE);
	char	refName[256];
	DataBuf	aBuf;
	aBuf.buf = (char *)refName;
	aBuf.len = 256;
	aRef->GetDiscriptionText(&aBuf);
	int name= strcmp((char *)info->componentName, (char *)refName);
	exists= (name == 0);
	return( exists );
}
